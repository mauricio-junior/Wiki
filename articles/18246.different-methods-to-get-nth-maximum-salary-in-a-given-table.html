--DIFFERENT METHODES TO GET N-TH MAXIMUM SALARY IN A TABLE.
-- METHODE-1
SELECT A.EMPNO,A.ENAME,A.SAL
 FROM EMP A JOIN EMP B ON A.SAL<=B.SAL
 GROUP BY A.SAL,A.EMPNO,A.ENAME
 HAVING COUNT(A.SAL)=4
 
 -- METHODE-2
  SELECT TOP 1 *
 FROM
 (
 SELECT TOP 4 * FROM EMP ORDER BY SAL DESC
 ) TEMP
 ORDER BY SAL ASC
 
 -- METHODE-3
 WITH CTE
 AS
 (
 SELECT *,ROW_NUMBER() OVER(ORDER BY SAL DESC) AS RNUM FROM EMP
 )
 SELECT * FROM CTE WHERE RNUM=4
 
 -- METHODE- 4
 SELECT * FROM EMP WHERE SAL=
 (
 SELECT MIN(SAL) AS SAL
 FROM
 (
 SELECT TOP 4 EMPNO,ENAME,HIREDATE,SAL FROM EMP ORDER BY SAL DESC
 ) TEMP
 )
 
 -- METHODE- 5
 SELECT * FROM EMP A
 WHERE
 (SELECT COUNT(*) FROM EMP WHERE SAL>=A.SAL)=4
